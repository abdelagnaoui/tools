// Format helper
function format(ms) {
  const totalSeconds = Math.floor(ms / 1000);
  const minutes = Math.floor(totalSeconds / 60);
  const seconds = totalSeconds % 60;
  return `${String(minutes).padStart(1, '0')}:${String(seconds).padStart(2, '0')}`;
}

// Update countdowns
function updateCountdowns() {
  remainingClickTime -= 1000;
  remainingStopTime -= 1000;

  if (remainingClickTime < 0) remainingClickTime = 0;
  if (remainingStopTime < 0) remainingStopTime = 0;

  clickTimerDisplay.textContent = `Next click in: ${format(remainingClickTime)}`;
  stopTimerDisplay.textContent = `Stopping in: ${format(remainingStopTime)}`;
}

// Start
function startScript() {
  clearAll();

  intervalMs = parseInt(intervalInput.value) * 1000;
  remainingClickTime = intervalMs;
  remainingStopTime = parseInt(durationInput.value) * 1000;

  clickIntervalId = setInterval(() => {
    if (elements[7]) elements[7].click();
    remainingClickTime = intervalMs;
  }, intervalMs);

  countdownIntervalId = setInterval(updateCountdowns, 1000);

  stopTimeout = setTimeout(() => {
    stopScript();
    clickTimerDisplay.textContent = "⏹ Clicks stopped";
    stopTimerDisplay.textContent = "✅ Done!";
  }, remainingStopTime);

  toggleButton.textContent = "⏹ Stop";
  running = true;
}

// Stop
function stopScript() {
  clearAll();
  toggleButton.textContent = "▶ Start";
  running = false;
}

// Clear all timers
function clearAll() {
  clearInterval(clickIntervalId);
  clearInterval(countdownIntervalId);
  clearTimeout(stopTimeout);
}

// Toggle on button click
toggleButton.addEventListener("click", () => {
  if (!running) {
    startScript();
  } else {
    stopScript();
    clickTimerDisplay.textContent = "⏹ Stopped manually";
    stopTimerDisplay.textContent = "⌛ Waiting...";
  }
});

  let stopFlag = false;
  let pauseFlag = false;
  let delay = 2000;

  // ===== PANEL =====
  const panel = document.createElement('div');
  panel.style.position = 'fixed';
  panel.style.top = '20px';
  panel.style.left = '20px';
  panel.style.zIndex = 9999;
  panel.style.width = '420px';
  panel.style.background = 'rgba(0,0,0,0.2)';
  panel.style.backdropFilter = 'blur(10px)'; // glass effect زوين 😎
  panel.style.padding = '15px';
  panel.style.borderRadius = '14px';
  panel.style.boxShadow = '0 10px 30px rgba(0,0,0,0.6)';
  panel.style.color = '#fff';
  panel.style.fontFamily = 'system-ui';
  document.body.appendChild(panel);

  // ===== TITLE =====
  const title = document.createElement('div');
  title.textContent = '⚡ Control Panel';
  title.style.fontSize = '16px';
  title.style.marginBottom = '10px';
  title.style.fontWeight = 'bold';
  panel.appendChild(title);

  // ===== TEXTAREA =====
  const inputArea = document.createElement('textarea');
  inputArea.placeholder = 'Paste your lines here...';
  inputArea.style.width = '100%';
  inputArea.style.height = '140px';
  inputArea.style.borderRadius = '10px';
  inputArea.style.padding = '10px';
  inputArea.style.marginBottom = '10px';
  inputArea.style.border = 'none';
  inputArea.style.outline = 'none';
  panel.appendChild(inputArea);

  // ===== DELAY ROW =====
  const delayRow = document.createElement('div');
  delayRow.style.display = 'flex';
  delayRow.style.alignItems = 'center';
  delayRow.style.gap = '10px';
  delayRow.style.marginBottom = '12px';

  const delayLabel = document.createElement('span');
  delayLabel.textContent = 'Lines:';
  delayLabel.style.fontSize = '13px';

  const delayInput = document.createElement('input');
  delayInput.type = 'number';
  delayInput.value = 2000;
  delayInput.style.flex = '1';
  delayInput.style.padding = '6px';
  delayInput.style.borderRadius = '8px';
  delayInput.style.border = 'none';

  delayInput.oninput = () => {
    delay = parseInt(delayInput.value) || 0;
  };

  delayRow.appendChild(delayLabel);
  delayRow.appendChild(delayInput);
  panel.appendChild(delayRow);

  // ===== BUTTON ROW =====
  const btnRow = document.createElement('div');
  btnRow.style.display = 'grid';
  btnRow.style.gridTemplateColumns = '1fr 1fr';
  btnRow.style.gap = '8px';

  const styleBtn = (btn, color) => {
    btn.style.padding = '10px';
    btn.style.border = 'none';
    btn.style.borderRadius = '10px';
    btn.style.cursor = 'pointer';
    btn.style.color = '#fff';
    btn.style.background = color;
    btn.style.fontWeight = 'bold';
  };

  const startButton = document.createElement('button');
  startButton.textContent = '▶ Start';
  styleBtn(startButton, '#22c55e');

  const pauseButton = document.createElement('button');
  pauseButton.textContent = '⏸ Pause';
  styleBtn(pauseButton, '#facc15');

  const resumeButton = document.createElement('button');
  resumeButton.textContent = '⏵ Resume';
  styleBtn(resumeButton, '#38bdf8');

  const stopButton = document.createElement('button');
  stopButton.textContent = '⛔ Stop';
  styleBtn(stopButton, '#ef4444');

  btnRow.appendChild(startButton);
  btnRow.appendChild(pauseButton);
  btnRow.appendChild(resumeButton);
  btnRow.appendChild(stopButton);

  panel.appendChild(btnRow);

  // ===== CONTROLS =====
  stopButton.onclick = () => stopFlag = true;
  pauseButton.onclick = () => pauseFlag = true;
  resumeButton.onclick = () => pauseFlag = false;

  const wait = (ms) => new Promise(r => setTimeout(r, ms));

  const waitForButtonReenable = async (button, timeout = 120000) => {
    const start = Date.now();
    while (button.disabled || button.classList.contains('ui-state-disabled')) {
      if (stopFlag) throw new Error('Stopped manually.');
      while (pauseFlag) await wait(300);
      if (Date.now() - start > timeout) throw new Error('Timeout.');
      await wait(200);
    }
  };

  startButton.onclick = async () => {
    stopFlag = false;
    pauseFlag = false;

    const lines = inputArea.value.split('\n').filter(l => l.trim());
    if (!lines.length) return alert('Paste lines first');

    const chunks = [];
    for (let i = 0; i < lines.length; i += 100) {
      chunks.push(lines.slice(i, i + 100).join('\n'));
    }

    const targetArea = document.getElementById('otherdoms');
    const button = document.querySelector('input[name="Pause"]');

    if (!targetArea || !button) return alert('Elements not found');

    for (let i = 0; i < chunks.length; i++) {
      if (stopFlag) return alert('Stopped');

      while (pauseFlag) await wait(300);

      targetArea.value = chunks[i];
      targetArea.dispatchEvent(new Event('input', { bubbles: true }));

      button.click();

      try {
        await waitForButtonReenable(button);
      } catch (err) {
        alert(err.message);
        return;
      }

      await wait(delay);
    }

    alert('Done ✅');
  };
})();panel.style
